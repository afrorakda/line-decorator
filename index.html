decorateBtn.addEventListener('click', () => { const text = input.value; const prefix = prefixInput.value; const suffix = suffixInput.value; if (text) { const lines = text.split('\n'); let decoratedCount = 0; const processed = lines.map(line => { if (line.trim() === '') return line; decoratedCount++; return `${prefix}${line}${suffix}`; }).join('\n'); // 【ここがポイント】入力欄の中身を結果に書き換える input.value = processed; navigator.clipboard.writeText(processed).then(() => { statusMsg.textContent = `✨ ${decoratedCount} lines decorated!`; statusMsg.style.opacity = '1'; // 視覚的なフィードバック input.classList.add('decorated'); const originalText = decorateBtn.textContent; decorateBtn.textContent = 'Decorated & Copied! ✓'; setTimeout(() => { decorateBtn.textContent = originalText; statusMsg.style.opacity = '0'; input.classList.remove('decorated'); }, 2000); }); } });
